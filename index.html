#include <iostream>
#include <numeric>

using namespace std;

inline void output(pair<long int, long int> f)
{
    if (!f.second)
        cout << "Inf";
    else
    {
        bool flag = f.first < 0 && (f.first *= -1), space = false;
        cout << (flag ? "(-" : "");
        if (f.first / f.second || f.first == 0) cout << f.first / f.second, space = true;
        if (f.first % f.second) cout << (space ? " " : "") << f.first % f.second << "/" << f.second;
        cout << (flag ? ")" : "");
    }
}

inline void simplify(pair<long int, long int> &f)
{
    long int g = gcd(f.first, f.second);
    f.first /= g, f.second /= g;
    if (f.second < 0) f.first *= -1, f.second *= -1;
}

int main()
{
    pair<long int, long int> a, b, c;
    scanf("%ld/%ld %ld/%ld", &a.first, &a.second, &b.first, &b.second);
    simplify(a), simplify(b);
    long int lab = lcm(a.second, b.second), na = a.first * lab / a.second, nb = b.first * lab / b.second;
    c.second = lab, c.first = na + nb, simplify(c);
    output(a), cout << " + ", output(b), cout << " = ", output(c), cout << endl;
    c.second = lab, c.first = na - nb, simplify(c);
    output(a), cout << " - ", output(b), cout << " = ", output(c), cout << endl;
    c.second = a.second * b.second, c.first = a.first * b.first, simplify(c);
    output(a), cout << " * ", output(b), cout << " = ", output(c), cout << endl;
    c.second = a.second * b.first, c.first = a.first * b.second, simplify(c);
    output(a), cout << " / ", output(b), cout << " = ", output(c), cout << endl;
}
