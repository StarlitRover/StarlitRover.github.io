#include <iostream>
#include <vector>
#include <bitset>

using namespace std;

int main()
{
    int n, m, a, b, k, x = -1;
    cin >> n >> m;
    vector<bitset<201>> edge(n + 1);
    for (int i = 0; i < m; ++i)
    {
        cin >> a >> b;
        edge[a][b] = edge[b][a] = true;
    }
    cin >> k;
    for (int i = 0; i < k; ++i, x = -1)
    {
        cin >> m;
        bitset<201> flag;
        bool ans = true;
        for (int j = 0; j < m; ++j)
        {
            cin >> a;
            if (!ans) continue;
            if ((flag & edge[a]) != flag) ans = false;
            flag[a] = true;
        }
        for (int j = 1; j <= n; ++j)
            if (!flag[j] && ((edge[j] & flag) == flag))
            {
                x = j;
                break;
            }
        if (ans && x == -1) printf("Area %d is OK.\n", i + 1);
        else if (ans) printf("Area %d may invite more people, such as %d.\n", i + 1, x);
        else printf("Area %d needs help.\n", i + 1);
    }
}
