#include <iostream>
#include <map>
#include <vector>

using namespace std;
map<int, pair<int, int>> sons;
map<int, pair<bool, bool>> has;

int insert(int r, int k)
{
    if (r < k && has[r].second)
        sons[r].second = insert(sons[r].second, k);
    else if (r < k)
        sons[r].second = k, has[r].second = true;
    else if (r > k)
    {
        sons[k].first = r, has[k].first = true;
        return k;
    }
    return r;
}

int main()
{
    int n, r, k;
    cin >> n >> r;
    for (int i = 1; i < n; ++i)
        cin >> k, r = insert(r, k);
    vector<int> crt{r}, nxt;
    bool space = false;
    while (!crt.empty())
    {
        for (auto &t: crt)
        {
            printf("%s%d", space ? " " : (space = true, ""), t);
            if (has[t].first) nxt.push_back(sons[t].first);
            if (has[t].second) nxt.push_back(sons[t].second);
        }
        crt = std::move(nxt), nxt = vector<int>();
    }
}
