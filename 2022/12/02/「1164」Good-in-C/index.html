#include <iostream>
#include <vector>

using namespace std;

int main()
{
    vector<vector<string>> matrix(26, vector(7, string()));
    for (int i = 0; i < 26; ++i)
        for (int j = 0; j < 7; ++j)
            getline(cin, matrix[i][j]);
    string s, w;
    getline(cin, s);
    s += " ";
    bool line = false;
    for (auto &c: s)
    {
        if (c >= 'A' && c <= 'Z')
            w += c;
        else if (!w.empty())
        {
            if (line) printf("\n");
            else line = true;
            for (int i = 0; i < 7; ++i)
                for (int j = 0; j < w.size(); ++j)
                    printf("%s%s", matrix[w[j] - 'A'][i].c_str(), j == w.size() - 1 ? "\n" : " ");
            w.clear();
        }
    }
}
