#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main()
{
    int v, e, a, b, d, k;
    cin >> v >> e;
    vector<vector<pair<int, int>>> edge(v + 1);
    for (int i = 0; i < e; ++i)
    {
        cin >> a >> b >> d;
        edge[a].emplace_back(b, d);
        edge[b].emplace_back(a, d);
    }
    cin >> k;
    for (int i = 0; i < k; ++i)
    {
        vector<int> dj(v), dist(v + 1, INT32_MAX);
        vector<bool> visited(v + 1, false);
        bool ans = true;
        for (auto &t: dj) cin >> t;
        for (auto &t: edge[dj[0]])
            dist[t.first] = t.second;
        visited[dj[0]] = true;
        for (int j = 1; j < v; dist[dj[j]] = INT32_MAX, visited[dj[j]] = true, ++j)
        {
            if (!(ans = *min_element(dist.begin(), dist.end()) == dist[dj[j]]))
                break;
            for (auto &t: edge[dj[j]])
            {
                if (visited[t.first]) continue;
                dist[t.first] = min(dist[dj[j]] + t.second, dist[t.first]);
            }
        }
        printf(ans ? "Yes\n" : "No\n");
    }
}
