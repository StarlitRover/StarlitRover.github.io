Input Specification:

Each input file contains one test case. Each case occupies 2 lines, and each line contains the information of a polynomial:

where K is the number of nonzero terms in the polynomial， and are the exponents and coefficients, respectively. It is given that ，.

Output Specification:

For each test case you should output the product of A and B in one line, with the same format as the input. Notice that there must be NO extra space at the end of each line. Please be accurate up to 1 decimal place.

2 1 2.4 0 3.2

#include <iostream>
#include <map>

using namespace std;

int main()
{
    map<int, double> a, b, p;
    int k, n;
    double m;
    cin >> k;
    //A input
    for (int i = 0; i < k; ++i)
    {
        cin >> n >> m;
        a[n] = m;
    }
    //B input
    cin >> k;
    for (int i = 0; i < k; ++i)
    {
        cin >> n >> m;
        b[n] = m;
    }
    //product
    for (auto &x: a)
        for (auto &y: b)
            p[x.first + y.first] += x.second * y.second;
    //delete monomial with coefficient 0
    for (auto z = p.begin(); z != p.end(); ++z)
        if (z->second == 0)
            p.erase(z);
    cout << p.size();
    for (auto z = p.rbegin(); z != p.rend(); ++z)
        printf(" %d %.1f", z->first, z->second);
}

#include<iostream>
#include<map>

using namespace std;

int main()
{
    map<int, double> x, y, z;
    int k, n;
    double a;
    cin >> k;
    for (int i = 0; i < k; ++i)
    {
        cin >> n >> a;
        x[n] = a;
    }
    cin >> k;
    for (int i = 0; i < k; ++i)
    {
        cin >> n >> a;
        y[n] = a;
    }

    for (auto &p: x)
        z[p.first] += p.second;
    for (auto &p: y)
    {
        z[p.first] += p.second;
        if (z[p.first] == 0)
            z.erase(p.first);
    }

    cout << z.size();
    for (auto s = z.rbegin(); s != z.rend(); ++s)
        printf(" %d %.1f", s->first, s->second);
}
