#include <iostream>
#include <vector>

using namespace std;

int main()
{
    int n, l, m, k;
    cin >> n >> l;
    vector<vector<int>> follow(n);
    for (int i = 0; i < n; ++i)
    {
        cin >> m;
        for (int j = 0; j < m; ++j)
        {
            cin >> k;
            follow[k - 1].push_back(i);
        }
    }
    cin >> m;
    vector<int> crt, nxt;
    for (int i = 0; i < m; ++i)
    {
        cin >> k;
        vector<bool> flag(n, true);
        flag[k - 1] = false;
        int cnt = 0;
        crt.push_back(k - 1);
        for (int j = 0; j < l && !crt.empty(); ++j)
        {
            for (auto &v: crt)
                for (auto &u: follow[v])
                    if (flag[u])
                    {
                        nxt.push_back(u);
                        flag[u] = false;
                        ++cnt;
                    }
            crt = std::move(nxt);
            nxt = vector<int>();
        }
        cout << cnt << endl;
    }
}
