#include <iostream>
#include <vector>
#include <algorithm>
#include <numeric>

using namespace std;
vector<int> sum, solve;
vector<vector<int>> score;

bool cmp(int &a, int &b)
{
    if (sum[a] != sum[b])
        return sum[a] > sum[b];
    else if (solve[a] != solve[b])
        return solve[a] > solve[b];
    else
        return a < b;
}

int main()
{
    int n, k, m, id, q, s;
    cin >> n >> k >> m;
    vector<int> mark(k), ids(n);
    vector<bool> flag(n, false);
    iota(ids.begin(), ids.end(), 0);
    sum = solve = vector<int>(n, 0);
    for (auto &v: mark)
        cin >> v;
    score = vector<vector<int>>(n, vector<int>(k, -1));
    for (int i = 0; i < m; ++i)
    {
        cin >> id >> q >> s;
        if (score[id - 1][q - 1] == -1 && s == -1)
            score[id - 1][q - 1] = 0;
        else
            score[id - 1][q - 1] = (score[id - 1][q - 1] < s ? (flag[id - 1] = true, s) : score[id - 1][q - 1]);
    }
    for (int i = 0; i < n; ++i)
        for (int j = 0; j < k; ++j)
        {
            sum[i] += (score[i][j] != -1) * score[i][j];
            solve[i] += score[i][j] == mark[j];
        }
    sort(ids.begin(), ids.end(), cmp);
    int rank = 1, pre = sum[ids[0]], cnt = 1;
    for (auto &v: ids)
    {
        if (!flag[v]) continue;
        rank = sum[v] == pre ? rank : cnt;
        printf("%d %05d %d", rank, v + 1, sum[v]);
        for (int i = 0; i < k; ++i)
            cout << " " << (score[v][i] == -1 ? "-" : to_string(score[v][i]));
        cout << endl;
        pre = sum[v];
        ++cnt;
    }
}
