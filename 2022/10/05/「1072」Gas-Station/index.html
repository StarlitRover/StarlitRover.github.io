#include <iostream>
#include <vector>
#include <map>
#include <algorithm>
#include <cmath>

using namespace std;

int main()
{
    int n, m, k, ds, p1, p2, d;
    cin >> n >> m >> k >> ds;
    vector<map<int, int>> dist(n + m + 1);
    string a, b;
    for (int i = 0; i < k; ++i)
    {
        cin >> a >> b >> d;
        p1 = a[0] == 'G' ? stoi(a.substr(1, a.size() - 1)) + n : stoi(a);
        p2 = b[0] == 'G' ? stoi(b.substr(1, b.size() - 1)) + n : stoi(b);
        dist[p1][p2] = dist[p2][p1] = d;
    }
    tuple<int, int, float> ans{0, 0, INT32_MAX};
    for (int i = n + 1; i < m + n + 1; ++i)
    {
        vector<int> min_dist(n + m + 1, INT32_MAX);
        vector<bool> isVisited(n + m + 1, false);
        isVisited[i] = true;
        for (auto &v: dist[i])
            min_dist[v.first] = v.second;
        int cnt = n, sum = 0;
        while (cnt)
        {
            int min = INT32_MAX, idx = -1;
            for (int j = 1; j < n + m + 1; ++j)
                if (!isVisited[j] && min_dist[j] < min)
                {
                    min = min_dist[j];
                    idx = j;
                }
            if (min > ds) break;
            for (auto &v: dist[idx])
                if (!isVisited[v.first] && v.second + min < min_dist[v.first])
                    min_dist[v.first] = v.second + min;
            isVisited[idx] = true;
            cnt -= (idx <= n ? (sum += min, 1) : 0);
        }
        if (!cnt)
        {
            int dmin = *min_element(min_dist.begin() + 1, min_dist.begin() + n + 1);
            float avg = 1.0 * sum / n;
            if (dmin > get<1>(ans) || dmin == get<1>(ans) && avg < get<2>(ans))
                ans = {i - n, dmin, avg};
        }
    }
    if (get<0>(ans))
        printf("G%d\n%.1f %.1f", get<0>(ans), float(get<1>(ans)), round(get<2>(ans) * 10) / 10);
    else
        printf("No Solution");
}
