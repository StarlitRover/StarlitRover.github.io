#include <iostream>
#include <vector>

using namespace std;
vector<pair<int, int>> son;
vector<int> in;

void in_order(int root)
{
    if (root == -1) return;
    in_order(son[root].second);
    in.push_back(root);
    in_order(son[root].first);
}

int main()
{
    int n;
    cin >> n;
    son.resize(n);
    string l, r;
    int root = (n - 1) * n / 2;
    for (auto &k: son)
    {
        cin >> l >> r;
        k.first = (l == "-" ? -1 : (root -= stoi(l), stoi(l)));
        k.second = (r == "-" ? -1 : (root -= stoi(r), stoi(r)));
    }
    cout << root;
    vector<int> crt{3}, nxt;
    do
    {
        for (auto k: crt)
        {
            if (son[k].first != -1) nxt.push_back(son[k].first);
            if (son[k].second != -1) nxt.push_back(son[k].second);
        }
        for (int i = int(nxt.size() - 1); i >= 0; --i)
            cout << " " << nxt[i];
        crt = std::move(nxt);
        nxt = vector<int>();
    } while (!crt.empty());
    cout << endl;
    in_order(root);
    cout << in[0];
    for (int i = 1; i < n; ++i)
        cout << " " << in[i];
}
