#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main()
{
    int n, turn;
    cin >> n;
    vector<int> org(n), crt(n);
    for (auto &k: org)
        cin >> k;
    for (auto &k: crt)
        cin >> k;
    turn = n;
    while (--turn >= 0 && crt[turn] == org[turn]);
    if (turn == -1)
    {
        turn = 0;
        while (turn < n && crt[turn] <= crt[turn + 1]) ++turn;
    }
    vector<int> copy(org);
    sort(copy.begin(), copy.begin() + turn + 1);
    bool isMerge = false;
    // Short-circuit evaluation
    for (int i = 0; i < n && (copy[i] == crt[i] || (isMerge = true , false)); ++i);
    if (isMerge)
    {
        int delta = 1;
        while (delta *= 2, crt != copy)
        {
            copy = org;
            for (int i = 0; i < n; i += delta)
                sort(org.begin() + i, org.begin() + (i + delta > n ? n : i + delta));
        }
    }
    else
        sort(org.begin(), org.begin() + turn + 2);
    cout << (isMerge ? "Merge Sort" : "Insertion Sort") << endl;
    for (int i = 0; i < n; ++i)
        cout << (i == 0 ? "" : " ") << org[i];
}
