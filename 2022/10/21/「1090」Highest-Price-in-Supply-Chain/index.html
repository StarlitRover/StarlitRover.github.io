using namespace std;

int max_d = 0, num = 0;
vector<vector<int>> rtl;

void dfs(int r, int d)
{
    if (d > max_d)
    {
        num = 1;
        max_d = d;
    }
    else if (d == max_d)
        num += 1;
    for (auto &k: rtl[r])
        dfs(k, d + 1);
}
