#include <iostream>
#include <stack>
#include <vector>
#include <algorithm>

using namespace std;
vector<int> post, in;

void PostOrder(int begin, int end)
{
    if (begin > end) return;
    int root = min_element(in.begin() + begin, in.begin() + end + 1) - in.begin();
    PostOrder(begin, root - 1);
    PostOrder(root + 1, end);
    post.push_back(in[root]);
}

int main()
{
    stack<int> s;
    int n, m, idx = 0;
    cin >> n;
    string opr;
    vector<int> v(n);
    for (int i = 0; i < 2 * n; ++i)
    {
        cin >> opr;
        if (opr == "Push")
        {
            cin >> v[idx];
            s.push(idx++);
        }
        else if (opr == "Pop")
        {
            in.push_back(s.top());
            s.pop();
        }
    }
    PostOrder(0, n - 1);
    for (int i = 0; i < n; ++i)
        cout << (i == 0 ? "" : " ") << v[post[i]];
}
