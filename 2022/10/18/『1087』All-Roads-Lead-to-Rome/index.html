#include <iostream>
#include <map>
#include <vector>

using namespace std;

vector<int> happy, nxt, path;
vector<vector<int>> pre;     // previous nodes in the least cost paths
vector<pair<int, int>> info; // <happiness sum, nodes num>

int collect(int &node)
{
    int sum = 0;
    for (auto k: pre[node])
    {
        auto p = make_pair(info[node].first + happy[k], info[node].second - 1);
        if (p > info[k])
        {
            info[k] = p;
            nxt[k] = node;
        }
        sum += (path[k] == -1 ? collect(k) : path[k]);
    }
    return path[node]=sum;
}

int main()
{
    int n, k, dst, c;
    scanf("%d %d", &n, &k);
    vector<string> city(n, string(3, 0));
    happy.resize(n, 0);
    map<string, int> num;
    cin >> city[0];
    num[city[0]] = 0;
    for (int i = 1; i < n; ++i)
    {
        scanf("%s %d", &city[i][0], &happy[i]);
        num[city[i]] = i;
        if (city[i] == "ROM")
            dst = i;
    }
    vector<vector<pair<int, int> > > cost(n);
    string c1(3, 0), c2(3, 0);
    for (int i = 0; i < k; ++i)
    {
        scanf("%s %s %d", &c1[0], &c2[0], &c);
        cost[num[c1]].emplace_back(num[c2], c);
        cost[num[c2]].emplace_back(num[c1], c);
    }
    vector<int> dist(n, INT32_MAX);
    vector<bool> done(n, false);
    dist[0] = 0, done[0] = true;
    pre.resize(n);
    for (auto &v: cost[0])
        pre[v.first].push_back(0), dist[v.first] = v.second;
    while (!done[dst])
    {
        int min_d = INT32_MAX;
        vector<int> opt;
        for (int i = 1; i < n; ++i)
        {
            if (done[i]) continue;
            else if (dist[i] < min_d)
            {
                opt.clear();
                opt.push_back(i);
                min_d = dist[i];
            }
            else if (dist[i] == min_d)
                opt.push_back(i);
        }
        for (auto &o: opt)
        {
            for (auto &v: cost[o])
                if (dist[o] + v.second < dist[v.first])
                {
                    dist[v.first] = dist[o] + v.second;
                    pre[v.first].clear();
                    pre[v.first].push_back(o);
                }
                else if (dist[o] + v.second == dist[v.first])
                    pre[v.first].push_back(o);
            done[o] = true;
        }
    }
    info.resize(n, {INT32_MIN, INT32_MIN});
    info[dst] = make_pair(happy[dst], 0);
    nxt = path = vector<int>(n, -1);
    path[0] = 1;
    printf("%d %d %d %d\n", collect(dst), dist[dst], info[0].first, info[0].first / -info[0].second);
    int idx = 0;
    printf("%s", city[0].c_str());
    while ((idx = nxt[idx]) != -1)
        printf("->%s", city[idx].c_str());
}
