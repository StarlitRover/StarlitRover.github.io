#include <iostream>
#include <vector>
#include <stack>
#include <tuple>

using namespace std;

int main()
{
    int m, n, l, t;
    cin >> m >> n >> l >> t;
    vector<vector<vector<int> > > pixel(l, vector(m, vector(n, 0)));

    for (int i = 0; i < l; ++i)
        for (int j = 0; j < m; ++j)
            for (int k = 0; k < n; ++k)
                cin >> pixel[i][j][k];
    int sum = 0;
    for (int i = 0; i < l; ++i)
        for (int j = 0; j < m; ++j)
            for (int k = 0; k < n; ++k)
            {
                if (!pixel[i][j][k]) continue;
                stack<tuple<int, int, int>> s;
                s.emplace(i, j, k);
                int piece = 0;
                while (!s.empty())
                {
                    int a = get<0>(s.top()), b = get<1>(s.top()), c = get<2>(s.top());
                    s.pop();
                    if (a < 0 || a >= l || b < 0 || b >= m || c < 0 || c >= n || !pixel[a][b][c]) continue;
                    pixel[a][b][c] = 0;
                    ++piece;
                    if (a > 0) s.emplace(a - 1, b, c);
                    if (a < l - 1) s.emplace(a + 1, b, c);
                    if (b > 0) s.emplace(a, b - 1, c);
                    if (b < m - 1) s.emplace(a, b + 1, c);
                    if (c > 0) s.emplace(a, b, c - 1);
                    if (c < n - 1) s.emplace(a, b, c + 1);
                }
                sum += piece >= t ? piece : 0;
            }
    cout << sum;
}
