#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;
vector<int> idx; // sort index
vector<pair<int, int>> son;

void make_idx(int r) // In-Order Traversal
{
    static int i = 0;
    if (son[r].first != -1)
        make_idx(son[r].first);
    idx[r] = i++;
    if (son[r].second != -1)
        make_idx(son[r].second);
}

int main()
{
    int n;
    cin >> n;
    son.resize(n);
    for (auto &s: son)
        cin >> s.first >> s.second;
    idx.resize(n);
    make_idx(0);
    vector<int> seq(n);
    for (auto &k: seq)
        cin >> k;
    sort(seq.begin(), seq.end());
    vector<int> crt{0}, nxt;
    cout << seq[idx[0]];
    do
    {
        for (auto &k: crt)
        {
            if (son[k].first != -1) nxt.push_back(son[k].first);
            if (son[k].second != -1) nxt.push_back(son[k].second);
        }
        for (auto &k: nxt)
            cout << " " << seq[idx[k]];
        crt = std::move(nxt);
        nxt = vector<int>();
    } while (!crt.empty());
}
