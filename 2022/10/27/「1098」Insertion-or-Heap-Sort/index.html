#include <iostream>

using namespace std;

inline void heap_down(vector<int> &heap, int r, int end)
{
    int max_r = (end - 1) / 2, j;
    while (r <= max_r)
    {
        if (2 * r + 2 <= end)
            j = (heap[2 * r + 2] > heap[2 * r + 1] ? 2 * r + 2 : 2 * r + 1);
        else if (2 * r + 1 <= end)
            j = 2 * r + 1;
        if (heap[j] > heap[r])
        {
            swap(heap[j], heap[r]);
            r = j;
        }
        else break;
    }
}
