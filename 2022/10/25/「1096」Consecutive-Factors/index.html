#include <iostream>
#include <cmath>

using namespace std;

int main()
{
    int n, k;
    cin >> n;
    int m, len, max_len = 0, start = n;
    for (int i = 2; i <= sqrt(n); ++i)
    {
        m = n, len = 0, k = i;
        while (m % k == 0)
        {
            m /= k++;
            ++len;
        }
        max_len = (len > max_len) ? (start = i, len) : max_len;
    }
    max_len = max_len == 0 ? 1 : max_len;
    printf("%d\n%d", max_len, start);
    for (int i = 1; i < max_len; ++i)
        printf("*%d", start + i);
}
