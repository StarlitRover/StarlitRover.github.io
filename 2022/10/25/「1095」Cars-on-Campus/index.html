#include <iostream>
#include <vector>
#include <set>
#include <map>
#include <algorithm>

using namespace std;

int main()
{
    int n, k, h, m, s, t;
    cin >> n >> k;
    map<string, set<pair<int, bool>>> record;
    string p(7, 0), st(3, 0);
    for (int i = 0; i < n; ++i)
    {
        scanf("%s %d:%d:%d %s", &p[0], &h, &m, &s, &st[0]);
        t = 3600 * h + 60 * m + s;
        record[p].emplace(t, st[0] == 'i');
    }
    vector<pair<int, bool>> stamp;
    map<string, int> time;
    for (auto &r: record)
    {
        int pre = -1;
        for (auto &w: r.second)
        {
            if (w.second) pre = w.first;
            else if (pre > -1)
            {
                stamp.emplace_back(pre, true);
                stamp.emplace_back(w.first, false);
                time[r.first] += (w.first - pre);
                pre = -1;
            }
        }
    }
    sort(stamp.begin(), stamp.end());
    int idx = 0, cnt = 0;
    for (int i = 0; i < k; ++i)
    {
        scanf("%d:%d:%d", &h, &m, &s);
        t = 3600 * h + 60 * m + s;
        while (idx < stamp.size() && stamp[idx].first <= t)
        {
            cnt += stamp[idx].second ? 1 : -1;
            ++idx;
        }
        printf("%d\n", cnt);
    }
    int max_t = 0;
    set<string> plt;
    for (auto &w: time)
    {
        if (w.second > max_t)
        {
            plt.clear();
            plt.emplace(w.first);
            max_t = w.second;
        }
        else if (w.second == max_t)
            plt.emplace(w.first);
    }
    for (auto &w: plt)
        printf("%s ", w.c_str());
    printf("%02d:%02d:%02d", max_t / 3600, max_t % 3600 / 60, max_t % 60);
}
