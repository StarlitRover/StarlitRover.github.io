#include <iostream>
#include <vector>
#include <algorithm>
#include <tuple>
#include <set>

using namespace std;

int main()
{
    int n, m, k;
    scanf("%d%d%d", &n, &m, &k);

    vector<int> quota(m);
    for (auto &q: quota)
        scanf("%d", &q);

    vector<tuple<int, int, int>> score(n);
    vector<vector<int>> pref(n, vector<int>(k));
    for (int i = 0; i < n; ++i)
    {
        scanf("%d%d", &get<1>(score[i]), &get<0>(score[i]));
        get<0>(score[i]) += get<1>(score[i]);
        get<2>(score[i]) = i;
        for (auto &p: pref[i])
            scanf("%d", &p);
    }

    sort(score.begin(), score.end(), greater<>());

    vector<bool> green(m, false);
    pair<int, int> pre{-1, -1};
    vector<set<int>> admit(m);
    for (auto &s: score)
    {
        if (pre.first != get<0>(s) || pre.second != get<1>(s))
            fill(green.begin(), green.end(), false);
        for (auto &p: pref[get<2>(s)])
            if (quota[p] > 0 || green[p])
            {
                quota[p] -= 1;
                green[p] = true;
                admit[p].insert(get<2>(s));
                break;
            }
        pre = make_pair(get<0>(s), get<1>(s));
    }

    for (auto &a: admit)
    {
        if (!a.empty())
        {
            printf("%d", *a.begin());
            for (auto aa = ++a.begin(); aa != a.end(); ++aa)
                printf(" %d", *aa);
        }
        printf("\n");
    }
}
