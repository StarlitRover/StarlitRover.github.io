A common implementation of a heap is the binary heap, in which the tree is a complete binary tree. (Quoted from Wikipedia at https://en.wikipedia.org/wiki/Heap_(data_structure))

One thing for sure is that all the keys along any path from the root to a leaf in a max/min heap must be in non-increasing/non-decreasing order.

#include <iostream>
#include <vector>

#define sign(x, y) (x==y?0:(x>y)+1)

using namespace std;
int n, jdg = 0;
vector<int> level, path;

void traverse(int r)
{
    if (r > n) return;
    path.push_back(level[r]);
    jdg |= sign(level[r], level[r / 2]);
    if (2 * r > n && 2 * r + 1 > n)
        for (int i = 0; i < path.size(); ++i)
            printf("%d%s", path[i], i < path.size() - 1 ? " " : "\n");
    traverse(2 * r + 1);
    traverse(2 * r);
    path.pop_back();
}

int main()
{
    cin >> n;
    level.resize(n + 1);
    for (int i = 1; i <= n; ++i)
        cin >> level[i];
    level[0] = level[1];
    traverse(1);
    printf("%s", jdg == 3 ? "Not Heap" : jdg == 2 ? "Min Heap" : "Max Heap");
}
