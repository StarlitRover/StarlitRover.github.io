#include <iostream>
#include <vector>

using namespace std;

int main()
{
    int v, e, a, b;
    cin >> v >> e;
    bitset<201> edge[v + 1];
    for (int i = 0; i < e; ++i)
    {
        cin >> a >> b;
        edge[a][b] = edge[b][a] = true;
    }
    int n, m;
    cin >> n;
    for (int i = 0; i < n; ++i)
    {
        cin >> m;
        bool ans = true;
        vector<bool> flag(v + 1, false);
        bitset<201> r(0);
        for (int j = 0; j < m; ++j)
        {
            cin >> a;
            if (!ans) continue;
            if ((edge[a] & r) != r) ans = false;
            r[a] = true, flag[a] = true;
        }
        if (!ans) printf("Not a Clique\n");
        else
        {
            for (int j = 1; j <= v; ++j)
            {
                if (flag[j]) continue;
                if ((edge[j] & r) == r)
                {
                    ans = false;
                    break;
                }
            }
            printf(ans ? "Yes\n" : "Not Maximal\n");
        }
    }
}
