#include <iostream>
#include <vector>

using namespace std;

int main()
{
    int n, m, a, b;
    cin >> n >> m;
    vector<vector<int>> edge(n + 1);
    for (int i = 0; i < m; ++i)
    {
        cin >> a >> b;
        edge[b].push_back(a);
    }
    cin >> m;
    vector<int> ans;
    for (int i = 0; i < m; ++i)
    {
        bool isTPG = true;
        vector<bool> flag(n + 1, false);
        for (int j = 0; j < n; ++j)
        {
            cin >> a, flag[a] = true;
            if (!isTPG) continue;
            for (auto &k: edge[a])
                if (!flag[k])
                {
                    isTPG = false;
                    break;
                }
        }
        if (!isTPG) ans.push_back(i);
    }
    cout << ans[0];
    for (int i = 1; i < ans.size(); ++i)
        cout << " " << ans[i];
}
