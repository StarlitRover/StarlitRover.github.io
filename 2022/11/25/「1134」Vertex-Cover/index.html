#include <iostream>
#include <bitset>
#include <vector>

using namespace std;

int main()
{
    int n, m, v, k, t, a, b;
    cin >> n >> m;
    vector<bitset<10000>> flag(n, 0);
    for (int i = 0; i < m; ++i)
    {
        cin >> a >> b;
        flag[a][i] = flag[b][i] = 1;
    }
    cin >> v;
    for (int i = 0; i < v; ++i)
    {
        cin >> k;
        bitset<10000> otc(0);
        for (int j = 0; j < k; ++j)
        {
            cin >> t;
            otc |= flag[t];
        }
        printf(otc.to_string().substr(10000 - m) == string(m, '1') ? "Yes\n" : "No\n");
    }
}
