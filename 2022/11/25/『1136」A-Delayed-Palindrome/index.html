#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
    string s, t, r;
    cin >> s, t = s;
    reverse(t.begin(), t.end());
    int c = 0, d, i;
    // ATTENTION: c=0 !!!
    for (i = 0; i < 10 && s != t; ++i, c = 0, r.clear())
    {
        for (int j = s.size() - 1; j >= 0; --j)
        {
            d = s[j] + t[j] - 2 * '0' + c;
            r.push_back(d % 10 + '0'), c = d / 10;
        }
        if (c) r.push_back('1');
        reverse(r.begin(), r.end());
        printf("%s + %s = %s\n", s.c_str(), t.c_str(), r.c_str());
        s = t = r;
        reverse(t.begin(), t.end());
    }
    if (i == 10) printf("Not found in 10 iterations.\n");
    else printf("%s is a palindromic number.\n", s.c_str());
}
