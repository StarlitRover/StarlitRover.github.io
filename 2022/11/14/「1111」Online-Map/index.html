#include <iostream>
#include <vector>
#include <tuple>

using namespace std;

int main()
{
    int n, m, v1, v2, ow, len, t, s, d;
    scanf("%d %d", &n, &m);
    vector<vector<tuple<int, int, int>>> info(n);
    for (int i = 0; i < m; ++i)
    {
        scanf("%d %d %d %d %d", &v1, &v2, &ow, &len, &t);
        info[v1].emplace_back(v2, len, t);
        if (!ow) info[v2].emplace_back(v1, len, t);
    }
    scanf("%d %d", &s, &d);
    vector<bool> isDtm(n, false);
    vector<int> dpre(n, -1);
    vector<pair<int, int>> dist(n, {INT32_MAX, INT32_MAX});
    pair<int, int> dopt{0, 0}, topt{0, 0};
    isDtm[s] = true, dist[s] = {0, 0};
    for (auto &k: info[s])
        dpre[get<0>(k)] = s, dist[get<0>(k)] = {get<1>(k), get<2>(k)};
    while (!isDtm[d])
    {
        int idx = -1;
        dopt = {INT32_MAX, INT32_MAX};
        for (int i = 0; i < n; ++i)
            if (!isDtm[i] && dist[i] < dopt)
                dopt = dist[i], idx = i;
        isDtm[idx] = true;
        for (auto &k: info[idx])
            if (make_pair(get<1>(k) + dopt.first, get<2>(k) + dopt.second) < dist[get<0>(k)])
            {
                dist[get<0>(k)] = make_pair(get<1>(k) + dopt.first, get<2>(k) + dopt.second);
                dpre[get<0>(k)] = idx;
            }
    }

    fill(isDtm.begin(), isDtm.end(), false);
    vector<int> tpre(n, -1);
    vector<pair<int, int>> time(n, {INT32_MAX, INT32_MAX});
    isDtm[s] = true, time[s] = {0, 0};
    for (auto &k: info[s])
        tpre[get<0>(k)] = s, time[get<0>(k)] = {get<2>(k), 1};
    while (!isDtm[d])
    {
        int idx = -1;
        topt = {INT32_MAX, INT32_MAX};
        for (int i = 0; i < n; ++i)
            if (!isDtm[i] && time[i] < topt)
                topt = time[i], idx = i;
        isDtm[idx] = true;
        for (auto &k: info[idx])
            if (make_pair(get<2>(k) + topt.first, topt.second + 1) < time[get<0>(k)])
            {
                time[get<0>(k)] = make_pair(get<2>(k) + topt.first, topt.second + 1);
                tpre[get<0>(k)] = idx;
            }
    }

    vector<int> pathd, patht;
    t = d;
    while (t != -1) pathd.push_back(t), t = dpre[t];
    t = d;
    while (t != -1) patht.push_back(t), t = tpre[t];
    if (pathd == patht)
    {
        printf("Distance = %d; Time = %d: %d", dopt.first, dopt.second, s);
        for (int i = pathd.size() - 2; i >= 0; --i)
            printf(" -> %d", pathd[i]);
    }
    else
    {
        printf("Distance = %d: %d", dopt.first, s);
        for (int i = pathd.size() - 2; i >= 0; --i)
            printf(" -> %d", pathd[i]);
        printf("\nTime = %d: %d", topt.first, s);
        for (int i = patht.size() - 2; i >= 0; --i)
            printf(" -> %d", patht[i]);
    }
}
