#include <iostream>

using namespace std;

int main()
{
    int n, cnt = 0;
    cin >> n;
    string t;
    double sum = 0;
    for (int i = 0; i < n; ++i)
    {
        cin >> t;
        int before = 0, after = 0;
        bool flag = true, point = false, sign = t[0] == '-' && (t.erase(0, 1), true);
        for (auto &c: t)
        {
            if (c == '.' && !point)
            {
                point = true;
                continue;
            }
            else if (!isdigit(c))
                flag = false;
            else if (!point)
                flag = (before += (before != 0 || c != '0')) < 5;
            else
                flag = ++after < 3;
            if (!flag) break;
        }
        if (flag && stod(t) <= 1000)
        {
            sum += (sign ? -1 : 1) * stod(t);
            cnt += 1;
        }
        else
        {
            t = (sign ? "-" : "") + t;
            printf("ERROR: %s is not a legal number\n", t.c_str());
        }
    }
    if (!cnt) printf("The average of 0 numbers is Undefined");
    else printf("The average of %d number%s is %.2f", cnt, cnt == 1 ? "" : "s", sum / cnt);
}
