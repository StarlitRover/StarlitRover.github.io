An AVL tree is a self-balancing binary search tree. In an AVL tree, the heights of the two child subtrees of any node differ by at most one; if at any time they differ by more than one, rebalancing is done to restore this property. Figures 1-4 illustrate the rotation rules.

Now given a sequence of insertions, you are supposed to output the level-order traversal sequence of the resulting AVL tree, and to tell if it is a complete binary tree.

Input Specification:

Each input file contains one test case. For each case, the first line contains a positive integer N (â‰¤Â 20). Then N distinct integer keys are given in the next line. All the numbers in a line are separated by a space.

Output Specification:

For each test case, insert the keys one by one into an initially empty AVL tree. Then first print in a line the level-order traversal sequence of the resulting AVL tree. All the numbers in a line must be separated by a space, and there must be no extra space at the end of the line. Then in the next line, printÂ YESÂ if the tree is complete, orÂ NOÂ if not.

88 70 61 63 65

Sample Output 1:

YES

#include <iostream>
#include <vector>
#include <map>

#define diff(root) (depth[sons[root].first]-depth[sons[root].second])
#define depth(root) (max(depth[sons[root].first],depth[sons[root].second])+1)

using namespace std;
map<int, pair<int, int>> sons;
map<int, int> depth;

void insert(int &root, int num)
{
    if (root == 0)
    {
        root = num, depth[num] = 1;
        return;
    }
    if (root > num) insert(sons[root].first, num);
    else insert(sons[root].second, num);
    depth[root] = depth(root);
    if (abs(diff(root)) < 2) return;
    bool isLeft = diff(root) > 0;
    int &so = isLeft ? sons[root].first : sons[root].second,
        &si = isLeft ? sons[root].second : sons[root].first,
        &sso = isLeft ? sons[so].first : sons[so].second,
        &ssi = isLeft ? sons[so].second : sons[so].first,
        tmp = so;
    // åˆ¤æ–­æ’å…¥èŠ‚ç‚¹æ˜¯å¦ä½äºå­æ ‘çš„åŒä¾§
    if ((diff(root) ^ diff(so)) > 0)
    {
        so = ssi, ssi = root, root = tmp;
        depth[ssi] = depth(ssi);
        depth[root] = depth(root);
    }
    else
    {
        int &ssso = isLeft ? sons[ssi].first : sons[ssi].second,
            &sssi = isLeft ? sons[ssi].second : sons[ssi].first,
            t1 = ssso, t2 = sssi;
        ssso = so, sssi = root, so = t2, root = ssi, ssi = t1;
        // æ³¨æ„é«˜åº¦æ›´æ–°é¡ºåºè‡ªä¸‹å¾€ä¸Š
        depth[ssso] = depth(ssso);
        depth[sssi] = depth(sssi);
        depth[root] = depth(root);
    }
}

int main()
{
    int n, m, r;
    cin >> n >> r;
    for (int i = 1; i < n; ++i)
    {
        cin >> m;
        insert(r, m);
    }
    bool flag = false;
    vector<pair<int, int>> crt{{r, 1}}, nxt;
    int end = -1;
    while (!crt.empty())
    {
        for (auto &k: crt)
        {
            printf("%s%d", flag ? " " : (flag = true, ""), k.first);
            if (sons[k.first].first)
                nxt.emplace_back(sons[k.first].first, 2 * k.second);
            if (sons[k.first].second)
                nxt.emplace_back(sons[k.first].second, 2 * k.second + 1);
            end = k.second;
        }
        crt = std::move(nxt);
        nxt = vector<pair<int, int>>();
    }
    printf(end == n ? "\nYES" : "\nNO");
}
