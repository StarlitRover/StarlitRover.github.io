#include <algorithm>

    int r, n, m;
    cin >> r;
    int seq[r];
    for (auto &k: seq)
        cin >> k;
    sort(seq, seq + r, greater());
    for (n = int(sqrt(r)); n > 1; --n)
        if (r % n == 0) break;
    m = r / n;
    int mtx[m][n], l = -1, idx = -1;
    while (++l < (n + 1) / 2)
    {
        for (int i = l; i < n - l; ++i)
            mtx[l][i] = seq[++idx];
        for (int i = l + 1; i < m - l; ++i)
            mtx[i][n - l - 1] = seq[++idx];
        for (int i = n - l - 2; i >= l; --i)
            mtx[m - l - 1][i] = seq[++idx];
        if (l != n / 2)
            for (int i = m - l - 2; i > l; --i)
                mtx[i][l] = seq[++idx];
    }
    for (int i = 0; i < m; ++i)
    {
        printf("%d", mtx[i][0]);
        for (int j = 1; j < n; ++j)
            printf(" %d", mtx[i][j]);
