#include <iostream>
#include <vector>

#define sign(x, y) (x==y?0:x>y?2:1)

using namespace std;
vector<int> level;

void printPost(int n)
{
    if (n >= level.size()) return;
    printPost(2 * n + 1);
    printPost(2 * n + 2);
    printf("%d%s", level[n], n == 0 ? "\n" : " ");
}

int main()
{
    int m, n, jdg = 0;
    cin >> m >> n;
    level.resize(n);
    bool flag = false;
    for (int i = 0; i < m; ++i, flag = false, jdg = 0)
    {
        cin >> level[0];
        for (int j = 1; j < n; ++j, flag = !flag)
        {
            cin >> level[j];
            jdg |= sign(level[j / 2 - flag], level[j]);
        }
        printf(jdg == 3 ? "Not Heap\n" : jdg == 2 ? "Max Heap\n" : "Min Heap\n");
        printPost(0);
    }
}
