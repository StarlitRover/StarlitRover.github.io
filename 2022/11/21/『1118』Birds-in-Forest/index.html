#include <iostream>
#include <vector>
#include <numeric>

using namespace std;

int main()

{
    int n, m, k, r, t = 0, b;
    cin >> n;
    vector<int> belong(1e4 + 1);
    vector<bool> flag(1e4 + 1, false), isRoot(1e4 + 1, false);
    iota(belong.begin(), belong.end(), 0);
    for (int i = 0; i < n; ++i)
    {
        cin >> m >> r, flag[r] = true;
        while (belong[r] != r) r = belong[r];
        for (int j = 1; j < m; ++j)
        {
            cin >> k, flag[k] = true;
            while (k != belong[k]) k = belong[k];
            belong[k] = r;
        }
    }
    for (int i = 0; i < 1e4 + 1; ++i)
    {
        if (!flag[i]) continue;
        k = belong[i], b = i;
        while (k != belong[k]) k = belong[k];
        if (!isRoot[k])
            isRoot[k] = true, ++t;
        belong[i] = k;
    }
    printf("%d %d\n", t, b);
    cin >> n;
    for (int i = 0; i < n; ++i)
    {
        cin >> m >> k;
        printf(belong[m] == belong[k] ? "Yes\n" : "No\n");
    }
