#include <iostream>
#include <set>
#include <map>
#include <vector>
#include <tuple>
#include <algorithm>

using namespace std;
typedef tuple<int, int, double, double> tiidd;

struct info
{
    set<int> edge;
    int est = 0, area = 0;
    bool flag = false;
};

int main()
{
    int n, id, f, m, k, c;
    scanf("%d", &n);
    map<int, info> p;
    for (int i = 0; i < n; ++i)
    {
        scanf("%d %d %d %d", &id, &f, &m, &k);
        if (f != -1)
            p[id].edge.insert(f), p[f].edge.insert(id);
        if (m != -1)
            p[id].edge.insert(m), p[m].edge.insert(id);
        for (int j = 0; j < k; ++j)
        {
            scanf("%d", &c);
            p[id].edge.insert(c);
            p[c].edge.insert(id);
        }
        scanf("%d %d", &p[id].est, &p[id].area);
    }

    vector<tiidd> ans;
    for (auto &r: p)
    {
        if (r.second.flag) continue;
        r.second.flag = true;
        set<int> crt{r.first}, nxt;
        int sum_est = r.second.est, sum_area = r.second.area, num = 1, min_id = r.first;
        while (!crt.empty())
        {
            for (auto &t: crt)
                for (auto &z: p[t].edge)
                {
                    if (p[z].flag) continue;
                    min_id = min(min_id, z);
                    p[z].flag = true;
                    ++num, sum_est += p[z].est, sum_area += p[z].area;
                    nxt.insert(z);
                }
            crt = std::move(nxt);
            nxt = set<int>();
        }
        ans.emplace_back(min_id, num, 1.0 * sum_est / num, 1.0 * sum_area / num);
    }
    sort(ans.begin(), ans.end(),
         [](tiidd &a, tiidd &b) { return get<3>(a) == get<3>(b) ? get<0>(a) < get<0>(b) : get<3>(a) > get<3>(b); });
    printf("%zu\n", ans.size());
    for (auto &t: ans)
        printf("%04d %d %.3f %.3f\n", get<0>(t), get<1>(t), get<2>(t), get<3>(t));
}
