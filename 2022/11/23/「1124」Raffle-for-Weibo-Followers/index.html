#include <iostream>
#include <set>
#include <vector>

using namespace std;

int main()
{
    int m, n, s;
    cin >> m >> n >> s;
    vector<string> nick(m);
    for (auto &t: nick)
        cin >> t;
    set<string> winner;
    for (int i = s - 1; i < m; i += n)
    {
        while (i < m && winner.find(nick[i]) != winner.end()) ++i;
        if (i != m)
        {
            winner.insert(nick[i]);
            cout << nick[i] << endl;
        }
    }
    if (m < s) cout << "Keep going...";
}
