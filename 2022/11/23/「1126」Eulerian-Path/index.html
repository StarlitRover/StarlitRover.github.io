In graph theory, an Eulerian path is a path in a graph which visits every edge exactly once. Similarly, an Eulerian circuit is an Eulerian path which starts and ends on the same vertex. They were first discussed by Leonhard Euler while solving the famous Seven Bridges of Konigsberg problem in 1736. It has been proven that connected graphs with all vertices of even degree have an Eulerian circuit, and such graphs are calledÂ Eulerian. If there are exactly two vertices of odd degree, all Eulerian paths start at one of them and end at the other. A graph that has an Eulerian path but not an Eulerian circuit is calledÂ semi-Eulerian. (Cited from https://en.wikipedia.org/wiki/Eulerian_path)

Given an undirected graph, you are supposed to tell if it is Eulerian, semi-Eulerian, or non-Eulerian.

#include <iostream>
#include <vector>

using namespace std;
vector<vector<int>> edge;
vector<bool> flag;

int dfs(int n)
{
    flag[n] = true;
    int sum = 0;
    for (auto &k: edge[n])
    {
        if (flag[k]) continue;
        sum += dfs(k);
    }
    return sum + 1;
}

int main()
{
    int n, m, a, b;
    cin >> n >> m;
    edge.resize(n + 1), flag.resize(n + 1, false);
    for (int i = 0; i < m; ++i)
    {
        cin >> a >> b;
        edge[a].push_back(b), edge[b].push_back(a);
    }
    int odd = edge[1].size() % 2, even = 1 - edge[1].size() % 2;
    cout << edge[1].size();
    for (int i = 2; i <= n; ++i)
    {
        cout << " " << edge[i].size();
        edge[i].size() % 2 ? ++odd : ++even;
    }
    cout << endl;
    if (dfs(1) < n) printf("Non-Eulerian");
    else printf(even == n ? "Eulerian" : (odd == 2 ? "Semi-Eulerian" : "Non-Eulerian"));
}
