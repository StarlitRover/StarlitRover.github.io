#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;
typedef pair<int, string> pis;

int main()
{
    int n, k;
    cin >> n >> k;
    vector<pis> info(n);
    for (auto &t: info)
        cin >> t.second >> t.first;
    sort(info.begin(), info.end(),
         [](pis &a, pis &b) { return a.first == b.first ? a.second < b.second : a.first > b.first; });
    vector<string> row;
    int r = n / k, u = r + n % r, head = 0, bias = 0;
    while (head < n)
    {
        row.resize(u);
        for (int i = 0; i < u; ++i)
        {
            row[u / 2 + bias] = info[head + i].second;
            bias = i % 2 ? -bias : -bias - 1;
        }
        cout << row[0];
        for (int j = 1; j < u; ++j)
            cout << " " << row[j];
        cout << endl;
        head += u, u = r, bias = 0, row.clear();
    }
}
