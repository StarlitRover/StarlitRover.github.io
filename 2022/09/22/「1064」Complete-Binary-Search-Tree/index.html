A Complete Binary Tree (CBT) is a tree that is completely filled, with the possible exception of the bottom level, which is filled from left to right.

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;
int n;
vector<int> level, keys;

void InOrder(int node)
{
    if (node >= n) return;
    static int idx = 0;
    InOrder(2 * node + 1);
    level[node] = keys[idx++];
    InOrder(2 * node + 2);
}

int main()
{
    cin >> n;
    keys.resize(n);
    level.resize(n);
    for (auto &k: keys)
        cin >> k;
    sort(keys.begin(), keys.end());
    InOrder(0);
    for (int i = 0; i < level.size(); ++i)
        cout << (i == 0 ? "" : " ") << level[i];
}
