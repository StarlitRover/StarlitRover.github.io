#include <iostream>

using namespace std;

class Float
{
public:
    string data;
    int bias = 0;

    explicit Float(int n)
    {
        cin >> data;
        size_t point = data.find('.');
        if (point == string::npos)
            point = data.size();
        else
            data.erase(data.begin() + point);

        for (char &it: data)
        {
            if (it != '0')
                break;
            ++bias;
        }
        data.erase(0, bias);
        if (data.empty())
        {
            data = "0";
            bias = 0;
        }
        else
            bias = point - bias;
        while (data.size() < n)
            data += "0";
    }

    inline string format(int n)
    {
        return "0." + data.substr(0, n) + "*10^" + to_string(bias);
    }
};

int main()
{
    int n;
    cin >> n;
    Float a(n), b(n);
    if (a.bias == b.bias && a.data.substr(0, n) == b.data.substr(0, n))
        cout << "YES " << a.format(n) << endl;
    else
        cout << "NO " << a.format(n) << " " << b.format(n) << endl;
}
