#include <iostream>
#include <map>

using namespace std;

int main()
{
    map<string, int> cnt;
    string s, word, ans;
    int max = -1;
    getline(cin, s);
    for (int i = 0; i < s.size(); ++i)
    {
        while (i < s.size() && !isalnum(s[i++]));
        i -= 1;
        for (; i < s.size() && isalnum(s[i]); ++i)
            word += tolower(s[i]);
        if (++cnt[word] > max)
        {
            max = cnt[word];
            ans = word;
        }
        else if (cnt[word] == max && word < ans)
            ans = word;
        word.clear();
    }
    cout << ans << " " << max;
}
