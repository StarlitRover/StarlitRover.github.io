import numpy as np
import pandas as pd
import joblib as jb

激活函数及导数定义

def Activater(x,option='sigmoid'):
    if option=='relu':
        return np.maximum(0,x)
    elif option=='sigmoid':
        return 1/(1+np.exp(-x))
    else:
        print("WRONG TYPE!!!")
        exit(0)

def Derivative(x,option='sigmoid'):
    if option=='relu':
        x[x>0] = 1
        x[x<=0] = 0
        return x
    elif option=='sigmoid':
        return np.exp(-x)/((1+np.exp(-x))**2)
    else:
        print("WRONG TYPE!!!")
        exit(0)

前向传播

def forward_propagation(X, W, b, option='sigmoid'):
    Z = []
    A = [X]
    for i in range(len(W)):
        Z.append(np.dot(W[i], A[i]) + b[i])
        A.append(Activater(Z[i],option))
    return Z, A

反向传播

def backward_propagation(Y, Z, A, W, b, option='sigmoid'):
    dZ = [[] for i in range(len(Z))]
    dW = [np.zeros(W[i].shape) for i in range(len(W))]
    db = [[] for i in range(len(b))]
    dA = [[] for i in range(len(A)-1)]
    dA[-1] = 2*(A[-1] - Y)
    for i in range(len(W) - 1, -1, -1):
        dZ[i] = dA[i] * Derivative(Z[i],option)
        for j in range(dZ[i].shape[1]):
            dW[i] += dZ[i][:,j:(j+1)].dot(A[i][:,j:(j+1)].T)
        dW[i] /= dZ[i].shape[1]
        db[i] = np.sum(dZ[i], axis=1, keepdims=True) / dZ[i].shape[1]
        if i>0:
            dA[i-1] = np.dot(W[i].T, dZ[i])
    return dW, db

误差函数



反向传播链式法则

def update_parameters(W, b, dW, db, learning_rate=1e-3):
    W = [W[i] - learning_rate * dW[i] for i in range(len(W))]
    b = [b[i] - learning_rate * db[i] for i in range(len(b))]
    return W, b

def Train(X, Y, W, b, epochs=300, option='sigmoid'):
    print("Training...")
    sum_10epoches=0
    for i in range(epochs):
        sum=0
        for j in range(len(X)):
            Z, A = forward_propagation(X[j], W, b, option)
            sum += np.sum(np.square(A[-1]-Y[j]))
            dW, db = backward_propagation(Y[j], Z, A, W, b, option)
            W, b = update_parameters(W, b, dW, db)

        sum_10epoches += sum/len(X)
        if (i+1) % 10 == 0:
            print("Epoch %d DONE , average loss: %f"%((i+1), sum_10epoches/10))
            sum_10epoches = 0
            # save outcome
            jb.dump(W, 'weight.pkl')
            jb.dump(b, 'b.pkl')

    return W, b

预测函数

def predict(X, W, b, option='sigmoid'):
    Z, A = forward_propagation(X, W, b, option)
    return [np.argmax(A[-1][:,i]) for i in range(A[-1].shape[1])]

for i in range(1,len(layers_dims)):
    W.append(np.random.randn(layers_dims[i], layers_dims[i - 1]) * np.sqrt(2 / layers_dims[i - 1]))
    b.append(np.zeros((layers_dims[i], 1)))
