Bézier Curve：交互增强

2022/4/1 计算机图形学 OpenGL 走码观花

☘︎ Preface

在很久很久的以前，我有介绍过Bézier Curve的理论推导以及基本实现过程（Bézier Curve：小试牛刀）。但不难看出，这个版本的Bézier Curve绘制程序的交互性极差。一旦确定就不能更改，控制点还要一个一个地输入坐标，每次运行还只能画一次，画完之后就什么操作都没了，有时候只想看个曲线却又去不掉花里胡哨的绘制过程（官方吐槽最为致命）。简单地说，除了画得好看简直一无是处。

在21世纪的今天，我们怎能容忍这种空有其表的程序呢！于是version 2.0横空出世。

☈ README.txt

Version 2.0在Version 1.0的基础上新增以下功能：

  1. 无需事先指定曲线的阶数，更无需输入控制点的坐标，直接在窗口内用鼠标左击确定控制点位置，点击后会出现相应的绿点，并输出相应的世界坐标；
  2. 在确定所有控制顶点位置后，鼠标右击绘制相应的贝塞尔曲线及绘制过程；
  3. 绘制完成后鼠标右击隐藏绘制过程，再次右击显示绘制过程；
  4. 绘制完成后按🆃可以修改值（默认为0.5）；
  5. 绘制完成后按住🅳可以拖动改变当前鼠标所在控制点的位置，同时贝塞尔曲线及绘制过程也会相应变化；
  6. 绘制完成后鼠标左击删除当前曲线及控制点，即清空画板回到第1步；
  7. 按␛退出程序.

⚖︎ Tips

  1. 对于实现点击选择控制顶点，首先我们需要将鼠标坐标转换为画布上的世界坐标（原点在中心，），函数void glfwGetCursorPos(GLFWwindow *window, double *xpos, double *ypos)得到的鼠标坐标(xpos,ypos)是一个像素坐标，原点位于窗口的左上角，y轴正方向向下，x轴正方向向右，根据窗口的宽高像素值WIDTH,HEIGHT以及简单的映射变换我们可以得到其世界坐标

    glfwGetCursorPos(window, &click_x, &click_y);
click_x = (click_x / WIDTH) * 2 - 1;
click_y = 1 - (click_y / HEIGHT) * 2;

  2. 对于贝塞尔曲线阶数的获取，我们只要在用户鼠标右击时查看控制顶点的个数即可

    order = control[0].size() - 1

  3. 至于如何使鼠标左击和右击时发挥不同的作用，其实是实现了一个简单的DFA。我设了两个布尔变量

    bool start = false, showLine = true;

    其中表示是否开始绘制曲线（用户在确定控制点位置时为false），表示是否展示绘制过程。我们将自动机的状态定义为有序对，其中1代表true，0代表false，那么其状态转移图可以表示为

  4. 清空画布就是上图中的情况

  5. 为全局变量，当start=true同时监听到按键🆃的按下时，将输入的值直接赋给，然后GeneratePointsViaT0()重新生成绘制过程

  6. 对于按键🅳的监听，只要在每个循环中查看🅳是否被按下，若处于按下状态，则计算距离当前鼠标最近的控制顶点，另外需要设定一个阈值（DRAG_SENSITIVITY，可以认为是拖动敏感度），只有最近距离小于阈值时才将该控制顶点移动到鼠标位置，并更新绘制过程与曲线细分点，详见void DragProcess(GLFWwindow *window)

  7. vector.resize()方法可以在保留原有数据的基础上扩充空间

C☺DE

#include <glew.h>
#include <glfw3.h>
#include <iostream>
#include <vector>

#define WIDTH 800
#define HEIGHT 600
#define CURVE_POINTS_NUM 100
#define DRAG_SENSITIVITY 0.01

using namespace std;

class Point
{
public:
    GLfloat x, y;

    Point(GLfloat _x, GLfloat _y) : x(_x), y(_y) {};

    inline Point operator*(const GLfloat &t) const
    {
        return {t * this->x, t * this->y};
    }

    inline Point operator+(const Point &rhs) const
    {
        return {this->x + rhs.x, this->y + rhs.y};
    }
};

int order;
bool start = false, showLine = true;
double click_x, click_y;
GLfloat t0 = 0.5;
vector<vector<Point>> control_points(1, vector<Point>()), middle_points;

void GeneratePointsViaT0()
{
    order = control_points[0].size() - 1;
    control_points.resize(order + 1);

    for (int i = 1; i < order + 1; ++i)
    {
        control_points[i].clear();
        for (int j = 0; j < order + 1 - i; ++j)
            control_points[i].emplace_back(control_points[i - 1][j] * (1 - t0) + control_points[i - 1][j + 1] * t0);
    }
}

void GenerateMiddlePoints()
{
    order = control_points[0].size() - 1;
    middle_points.clear();
    middle_points.resize(order + 1);
    middle_points[0] = control_points[0];

    for (int k = 0; k < CURVE_POINTS_NUM + 1; ++k)
    {
        GLfloat t = k / (1.0 * CURVE_POINTS_NUM);
        for (int i = 1; i < order; ++i)
        {
            middle_points[i].clear();
            for (int j = 0; j < order + 1 - i; ++j)
                middle_points[i].emplace_back(
                        middle_points[i - 1][j] * (1 - t) + middle_points[i - 1][j + 1] * t);
        }
        middle_points[order].emplace_back(middle_points[order - 1][0] * (1 - t) + middle_points[order - 1][1] * t);
    }
}

void MouseButtonCallback(GLFWwindow *window, int button, int action, int mods)
{
    if (action == GLFW_PRESS)
        switch (button)
        {
            case GLFW_MOUSE_BUTTON_LEFT:
                if (!start)
                {
                    glfwGetCursorPos(window, &click_x, &click_y);
                    click_x = (click_x / WIDTH) * 2 - 1;
                    click_y = 1 - (click_y / HEIGHT) * 2;
                    control_points[0].emplace_back(Point(click_x, click_y));
                    cout << "(" << click_x << " " << click_y << ")" << endl;
                }
                else
                {
                    order = 0;
                    control_points = move(vector<vector<Point>>(1, vector<Point>()));
                    start = false;
                }
                break;
            case GLFW_MOUSE_BUTTON_RIGHT:
                if (!start)
                {
                    if (control_points[0].size() <= 1)
                    {
                        cerr << "Control points are not enough!" << endl;
                        break;
                    }
                    GeneratePointsViaT0();
                    GenerateMiddlePoints();

                    start = true;
                    cout << endl;
                }
                else
                    showLine = !showLine;
                break;
        }
}

void KeyCallback(GLFWwindow *window, int key, int scancode, int action, int mode)
{
    if (action == GLFW_PRESS)
        switch (key)
        {
            case GLFW_KEY_ESCAPE:
                glfwSetWindowShouldClose(window, GL_TRUE);
                break;
            case GLFW_KEY_T:
                if (start)
                {
//                    showLine = false;
                    cout << "t0(∈(0,1)) = ";
                    cin >> t0;
                    GeneratePointsViaT0();
//                    showLine = true;
                }
                break;
        }
}

void DragProcess(GLFWwindow *window)
{
    if (start && glfwGetKey(window, GLFW_KEY_D) == GLFW_PRESS)
    {
        int closest_idx = -1;
        double dist, min_dist = 100;
        glfwGetCursorPos(window, &click_x, &click_y);
        click_x = (click_x / WIDTH) * 2 - 1;
        click_y = 1 - (click_y / HEIGHT) * 2;
        for (int i = 0; i < order + 1; ++i)
        {
            dist = (click_x - control_points[0][i].x) * (click_x - control_points[0][i].x) +
                   (click_y - control_points[0][i].y) * (click_y - control_points[0][i].y);
            if (dist < min_dist && dist < DRAG_SENSITIVITY)
            {
                min_dist = dist;
                closest_idx = i;
            }
        }
        control_points[0][closest_idx] = move(Point(click_x, click_y));
        GeneratePointsViaT0();
        GenerateMiddlePoints();
    }
}

int main()
{
    if (!glfwInit())
        return -1;
    GLFWwindow *window = glfwCreateWindow(WIDTH, HEIGHT, "Bézier Curve", NULL, NULL);
    if (!window)
        glfwTerminate();
    glfwMakeContextCurrent(window);
    glfwSetKeyCallback(window, KeyCallback);
    glfwSetMouseButtonCallback(window, MouseButtonCallback);

    while (!glfwWindowShouldClose(window))
    {
        glfwPollEvents();

        DragProcess(window);

        glClearColor(0.2, 0.1, 0.2, 1);
        glClear(GL_COLOR_BUFFER_BIT);

        if (start)
        {
            if (showLine)
            {
                glLineWidth(5);
                for (int i = 0; i < order; ++i)
                {
                    glColor3f(1.0 * (order - 1 - i) / (order - 1), 1.0 * i / (order - 1), 0);
                    glBegin(GL_LINE_STRIP);
                    for (auto &p: control_points[i])
                        glVertex2f(p.x, p.y);
                    glEnd();
                }

                glColor3f(0, 0.7, 0.7);
                glPointSize(10);
                glEnable(GL_POINT_SMOOTH);
                glBegin(GL_POINTS);
                glVertex2f(control_points[order][0].x, control_points[order][0].y);
                glEnd();
            }

            for (int i = 0; i < control_points[0].size(); ++i)
            {
                glColor3f(0, 1, 0);
                glPointSize(15);
                glBegin(GL_POINTS);
                glVertex2f(control_points[0][i].x, control_points[0][i].y);
                glEnd();
            }

            glColor3f(0, 0, 1);
            glBegin(GL_LINE_STRIP);
            for (auto &p: middle_points[order])
                glVertex2f(p.x, p.y);
            glEnd();
        }
        else
        {
            for (int i = 0; i < control_points[0].size(); ++i)
            {
                glColor3f(0, 1, 0);
                glPointSize(15);
                glEnable(GL_POINT_SMOOTH);
                glBegin(GL_POINTS);
                glVertex2f(control_points[0][i].x, control_points[0][i].y);
                glEnd();
            }
        }

        glfwSwapBuffers(window);
    }

    glfwTerminate();
    return 0;
}
